# Discord Bot設定
DISCORD_TOKEN=あなたのBotトークンを入力
GUILD_ID=あなたのサーバーIDを入力（任意）
OWNER_ID=あなたのユーザーIDを入力

# チャンネル設定
CONSOLE_OUTPUT_CHANNEL_ID=コンソール出力用チャンネルID（任意）
WELCOME_CHANNEL_ID=ウェルカムメッセージ用チャンネルID（任意）

# API設定
TENOR_API_KEY=LIVDSRZULELA
HUGGINGFACE_API_KEY=Hugging Face APIキー（無料AI機能使用時、任意）
UNSPLASH_API_KEY=Unsplash APIキー（無料画像機能使用時、任意）

# 音声・動画設定
FFMPEG_PATH=ffmpegのパス（例: C:/ffmpeg/bin/ffmpeg.exe）
VOICEVOX_PATH=VoiceVoxのパス（例: C:/Program Files/VOICEVOX/run.exe）
VOICEVOX_API_URL=http://127.0.0.1:50021
